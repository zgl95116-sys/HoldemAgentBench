# HoldemAgentBench Phase 1 MVP — Implementation Plan

> **For agentic workers:** 用 superpowers:executing-plans 逐任务执行。Steps 用 `- [ ]` 跟踪。

**Goal:** 用单条命令 `hab run quickstart --models <m1>,<m2>` 跑通 100 手 HU NLHE 对局，自动管理子进程、写入 hand history、不留孤儿进程。

**Architecture:** 单进程 Python orchestrator 内嵌 FastAPI shim 和 pokerkit 引擎；每个决策按需 fork 一个 `claude -p` 子进程，通过文件系统传递 game state / action；进程组隔离保证可清理。

**Tech Stack:** Python 3.11+, pokerkit, fastapi+uvicorn, httpx, typer (CLI), rich (TUI), pytest, pytest-asyncio

**Scope (Phase 1 MVP only):**
- 包骨架 + CLI 入口
- HU (heads-up) NLHE 引擎
- OpenRouter shim（非流式）
- Agent pool（subprocess 启停 + 超时 fold）
- 最小 CLAUDE.md 模板
- `hab init` + `hab run quickstart`
- E2E dry-run 测试（mock 模型，无真实 API 调用）

**Out of scope (后续 Phase):** MCP server / 7 工具 / 6-max / subagent / 评分系统 / dashboard / GitHub Pages。

```python
"""Anthropic /v1/messages <-> OpenAI /v1/chat/completions format translator."""
from __future__ import annotations

import json
from typing import Any


def _flatten_content(content: Any) -> str | list[dict]:
    """Anthropic content blocks -> OpenAI content. Returns string for plain text,
    or a list of OpenAI-style content parts when richer (e.g. tool results)."""
    if isinstance(content, str):
        return content
    if isinstance(content, list):
        # If purely text blocks, collapse to single string
        if all(b.get("type") == "text" for b in content):
            return "".join(b.get("text", "") for b in content)
        # Otherwise return as-is (caller handles tool_result etc.)
        return content
    return ""


def anthropic_request_to_openai(req: dict) -> dict:
    out: dict = {
        "model": req["model"],
        "max_tokens": req.get("max_tokens", 4096),
    }
    messages: list[dict] = []
    if req.get("system"):
        messages.append({"role": "system", "content": req["system"]})

    for msg in req.get("messages", []):
        role = msg["role"]
        content = msg.get("content", "")
        if isinstance(content, list):
            tool_calls: list[dict] = []
            text_parts: list[str] = []
            tool_results: list[tuple[str, str]] = []
            for block in content:
                btype = block.get("type")
                if btype == "text":
                    text_parts.append(block.get("text", ""))
                elif btype == "tool_use":
                    tool_calls.append({
                        "id": block["id"],
                        "type": "function",
                        "function": {
                            "name": block["name"],
                            "arguments": json.dumps(block.get("input", {})),
                        },
                    })
                elif btype == "tool_result":
                    tc_id = block["tool_use_id"]
                    tc_content = block.get("content", "")
                    if isinstance(tc_content, list):
                        tc_content = "".join(b.get("text", "") for b in tc_content if b.get("type") == "text")
                    tool_results.append((tc_id, tc_content))
            text = "".join(text_parts) if text_parts else None
            if tool_calls:
                m: dict = {"role": role, "content": text}
                m["tool_calls"] = tool_calls
                messages.append(m)
            elif tool_results:
                # Each tool_result becomes its own role=tool message
                for tc_id, tc_content in tool_results:
                    messages.append({"role": "tool", "tool_call_id": tc_id, "content": tc_content})
            else:
                messages.append({"role": role, "content": text or ""})
        else:
            messages.append({"role": role, "content": content})

    out["messages"] = messages

    # Tools
    if req.get("tools"):
        out["tools"] = [
            {
                "type": "function",
                "function": {
                    "name": t["name"],
                    "description": t.get("description", ""),
                    "parameters": t.get("input_schema", {"type": "object"}),
                },
            }
            for t in req["tools"]
        ]
    if req.get("temperature") is not None:
        out["temperature"] = req["temperature"]
    return out


_FINISH_REASON_MAP = {
    "stop": "end_turn",
    "length": "max_tokens",
    "tool_calls": "tool_use",
    "content_filter": "stop_sequence",
}


def openai_response_to_anthropic(resp: dict) -> dict:
    choice = resp["choices"][0]
    msg = choice["message"]
    finish = choice.get("finish_reason", "stop")

    content_blocks: list[dict] = []
    if msg.get("content"):
        content_blocks.append({"type": "text", "text": msg["content"]})
    for tc in msg.get("tool_calls") or []:
        try:
            args = json.loads(tc["function"]["arguments"])
        except (json.JSONDecodeError, TypeError):
            args = {}
        content_blocks.append({
            "type": "tool_use",
            "id": tc["id"],
            "name": tc["function"]["name"],
            "input": args,
        })

    usage = resp.get("usage", {})
    return {
        "id": resp.get("id", ""),
        "type": "message",
        "role": "assistant",
        "model": resp.get("model", ""),
        "content": content_blocks,
        "stop_reason": _FINISH_REASON_MAP.get(finish, "end_turn"),
        "stop_sequence": None,
        "usage": {
            "input_tokens": usage.get("prompt_tokens", 0),
            "output_tokens": usage.get("completion_tokens", 0),
        },
    }
```

```python
"""Decide whether a request goes to OpenRouter or directly to Anthropic."""
from __future__ import annotations


def route_request(model: str, anthropic_key: str | None) -> str:
    """Return 'anthropic_direct' if we should use the Anthropic API directly,
    otherwise 'openrouter'."""
    if model.startswith("anthropic/") and anthropic_key:
        return "anthropic_direct"
    return "openrouter"


def normalize_model_for_provider(model: str, provider: str) -> str:
    """Anthropic API expects e.g. 'claude-opus-4-7' not 'anthropic/claude-opus-4-7'."""
    if provider == "anthropic_direct" and model.startswith("anthropic/"):
        return model[len("anthropic/"):]
    return model
```

```python
"""Local FastAPI server that pretends to be the Anthropic /v1/messages endpoint."""
from __future__ import annotations

import asyncio
import logging
import socket
from typing import Any

import httpx
import uvicorn
from fastapi import FastAPI, Request

from hab.shim.router import route_request, normalize_model_for_provider
from hab.shim.translator import anthropic_request_to_openai, openai_response_to_anthropic

logger = logging.getLogger(__name__)

OPENROUTER_URL = "https://openrouter.ai/api/v1/chat/completions"
ANTHROPIC_URL = "https://api.anthropic.com/v1/messages"


class ShimServer:
    def __init__(
        self,
        openrouter_key: str | None,
        anthropic_key: str | None = None,
        host: str = "127.0.0.1",
        port: int | None = None,
    ):
        self.openrouter_key = openrouter_key
        self.anthropic_key = anthropic_key
        self.host = host
        self.port = port if port is not None else self._find_free_port()
        self.app = FastAPI(title="HAB Shim")
        self._setup_routes()
        self.server: uvicorn.Server | None = None
        self.task: asyncio.Task | None = None
        self._client = httpx.AsyncClient(timeout=120.0)

    @staticmethod
    def _find_free_port() -> int:
        with socket.socket() as s:
            s.bind(("", 0))
            return s.getsockname()[1]

    @property
    def base_url(self) -> str:
        return f"http://{self.host}:{self.port}"

    def _setup_routes(self):
        @self.app.post("/v1/messages")
        async def messages(request: Request):
            body: dict[str, Any] = await request.json()
            model = body.get("model", "")
            provider = route_request(model, self.anthropic_key)
            normalized = normalize_model_for_provider(model, provider)

            if provider == "anthropic_direct":
                body["model"] = normalized
                resp = await self._client.post(
                    ANTHROPIC_URL,
                    headers={
                        "x-api-key": self.anthropic_key or "",
                        "anthropic-version": "2023-06-01",
                        "content-type": "application/json",
                    },
                    json=body,
                )
                return resp.json()

            openai_req = anthropic_request_to_openai(body)
            resp = await self._client.post(
                OPENROUTER_URL,
                headers={
                    "Authorization": f"Bearer {self.openrouter_key}",
                    "HTTP-Referer": "https://github.com/holdem-agent-bench",
                    "X-Title": "HoldemAgentBench",
                    "content-type": "application/json",
                },
                json=openai_req,
            )
            data = resp.json()
            if "choices" not in data:
                logger.warning("OpenRouter returned non-standard response: %s", data)
                return data
            return openai_response_to_anthropic(data)

        @self.app.get("/healthz")
        async def healthz():
            return {"status": "ok", "port": self.port}

    async def start(self):
        config = uvicorn.Config(self.app, host=self.host, port=self.port, log_level="warning")
        self.server = uvicorn.Server(config)
        self.task = asyncio.create_task(self.server.serve())
        # Wait for server to be ready
        for _ in range(50):
            await asyncio.sleep(0.1)
            if self.server.started:
                return
        raise RuntimeError("Shim server failed to start")

    async def stop(self):
        if self.server is not None:
            self.server.should_exit = True
        if self.task is not None:
            try:
                await asyncio.wait_for(self.task, timeout=5.0)
            except asyncio.TimeoutError:
                self.task.cancel()
        await self._client.aclose()
```

```python
"""Action and LegalAction models, plus a validate_action() helper."""
from __future__ import annotations

from typing import Literal

from pydantic import BaseModel, Field

ActionType = Literal["fold", "check", "call", "bet", "raise", "all_in"]


class LegalAction(BaseModel):
    type: ActionType
    amount: float | None = None        # for call/check it's the exact amount (None for check/fold)
    amount_min: float | None = None    # for bet/raise
    amount_max: float | None = None


class Action(BaseModel):
    action: ActionType
    amount: float | None = None
    reason: str | None = None
    tool_calls_used: list[str] = Field(default_factory=list)
    timestamp: str | None = None
    hand_id: str | None = None


def validate_action(action: Action, legal: list[LegalAction]) -> str | None:
    """Return None if valid, otherwise an error string."""
    matching = [la for la in legal if la.type == action.action]
    if not matching:
        return f"action type '{action.action}' not legal; legal: {[la.type for la in legal]}"
    la = matching[0]
    if la.type in ("fold", "check"):
        return None
    if la.type == "call":
        if action.amount is not None and la.amount is not None and abs(action.amount - la.amount) > 1e-6:
            return f"call amount {action.amount} does not match required {la.amount}"
        return None
    if la.type in ("bet", "raise", "all_in"):
        if action.amount is None:
            return f"{la.type} requires amount"
        if la.amount_min is not None and action.amount < la.amount_min - 1e-6:
            return f"{la.type} amount {action.amount} below min {la.amount_min}"
        if la.amount_max is not None and action.amount > la.amount_max + 1e-6:
            return f"{la.type} amount {action.amount} above max {la.amount_max}"
        return None
    return None
```

```python
"""Game master: thin orchestration over pokerkit + event stream."""
from __future__ import annotations

import asyncio
import random
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import AsyncIterator, Any

from pokerkit import (
    NoLimitTexasHoldem,
    Automation,
    State,
)
from pydantic import BaseModel

from hab.engine.actions import Action, LegalAction, validate_action
from hab.engine.state import (
    ActionHistoryEntry,
    GameView,
    HandResult,
    HoleCards,
    Stacks,
)


@dataclass
class GameMasterConfig:
    players: list[str]
    small_blind: float = 1.0
    big_blind: float = 2.0
    starting_stack: float = 200.0
    hands_target: int = 100
    seed: int | None = None
    decision_timeout_sec: float = 300.0


class Event(BaseModel):
    type: str
    hand_id: str | None = None
    player_id: str | None = None
    game_view: GameView | None = None
    hole_cards: HoleCards | None = None
    legal_actions: list[LegalAction] = []
    payload: dict[str, Any] = {}


_STREET_NAMES = {0: "preflop", 1: "flop", 2: "turn", 3: "river"}


def _format_card(c) -> str:
    """pokerkit Card -> 'As' style. pokerkit Card has rank.value and suit.value."""
    return f"{c.rank.value}{c.suit.value}"


class GameMaster:
    def __init__(self, config: GameMasterConfig):
        self.config = config
        self.rng = random.Random(config.seed)
        self.stacks = {p: float(config.starting_stack) for p in config.players}
        self._action_received = asyncio.Event()
        self._submitted_action: Action | None = None
        self._submitting_player: str | None = None
        self._hand_counter = 0
        self.history: list[HandResult] = []
        self._button_idx = 0  # rotates each hand

    async def submit_action(self, player_id: str, action: Action) -> None:
        if self._submitting_player and player_id != self._submitting_player:
            raise RuntimeError(
                f"action from {player_id} but {self._submitting_player} is to act"
            )
        self._submitted_action = action
        self._action_received.set()

    async def events(self) -> AsyncIterator[Event]:
        yield Event(type="session_start", payload={"players": self.config.players})

        for hand_num in range(self.config.hands_target):
            self._hand_counter += 1
            hand_id = f"h_{self._hand_counter:05d}"
            async for ev in self._run_hand(hand_id):
                yield ev
            # rotate button after each hand
            self._button_idx = (self._button_idx + 1) % len(self.config.players)
            # if anyone busted, stop
            if any(s <= 0 for s in self.stacks.values()):
                break

        yield Event(
            type="session_complete",
            payload={
                "final_stacks": dict(self.stacks),
                "hands_played": self._hand_counter,
                "history": [h.model_dump() for h in self.history],
            },
        )

    async def _run_hand(self, hand_id: str) -> AsyncIterator[Event]:
        cfg = self.config
        # Determine player order: button acts first preflop in HU
        ordered = [
            cfg.players[(self._button_idx + i) % len(cfg.players)]
            for i in range(len(cfg.players))
        ]
        # pokerkit expects starting stacks as a sequence in seat order
        seat_stacks = [self.stacks[p] for p in ordered]

        state: State = NoLimitTexasHoldem.create_state(
            automations=(
                Automation.ANTE_POSTING,
                Automation.BET_COLLECTION,
                Automation.BLIND_OR_STRADDLE_POSTING,
                Automation.CARD_BURNING,
                Automation.HOLE_DEALING,
                Automation.BOARD_DEALING,
                Automation.HAND_KILLING,
                Automation.CHIPS_PUSHING,
                Automation.CHIPS_PULLING,
            ),
            ante_trimming_status=True,
            raw_antes=0,
            raw_blinds_or_straddles=(cfg.small_blind, cfg.big_blind),
            min_bet=cfg.big_blind,
            raw_starting_stacks=seat_stacks,
            player_count=len(ordered),
        )

        action_history: list[ActionHistoryEntry] = []

        yield Event(type="hand_start", hand_id=hand_id, payload={"button": ordered[0]})

        while state.status:
            if state.actor_index is None:
                continue  # automated step somehow returned mid-loop
            actor = ordered[state.actor_index]
            view = self._build_view(hand_id, ordered, state, action_history)
            hole_idx = state.actor_index
            # pokerkit stores hole cards under state.hole_cards[seat]
            hole = state.hole_cards[hole_idx] if hole_idx < len(state.hole_cards) else []
            hc = HoleCards(hand_id=hand_id, cards=[_format_card(c) for c in hole])

            self._action_received.clear()
            self._submitted_action = None
            self._submitting_player = actor

            yield Event(
                type="action_needed",
                hand_id=hand_id,
                player_id=actor,
                game_view=view,
                hole_cards=hc,
                legal_actions=view.legal_actions,
            )

            try:
                await asyncio.wait_for(
                    self._action_received.wait(),
                    timeout=self.config.decision_timeout_sec,
                )
            except asyncio.TimeoutError:
                self._submitted_action = Action(action="fold", hand_id=hand_id, reason="timeout")

            action = self._submitted_action or Action(action="fold", hand_id=hand_id, reason="no_action")
            err = validate_action(action, view.legal_actions)
            if err:
                action = Action(action="fold", hand_id=hand_id, reason=f"invalid_action:{err}")

            self._apply_action(state, action, view.legal_actions)
            action_history.append(ActionHistoryEntry(
                player_id=actor,
                street=view.street,
                action=action.action,
                amount=action.amount,
            ))

        # Hand complete -> compute deltas
        new_stacks = {ordered[i]: float(state.stacks[i]) for i in range(len(ordered))}
        deltas = {p: new_stacks[p] - self.stacks[p] for p in ordered}
        self.stacks = new_stacks
        winner = max(deltas, key=deltas.get) if deltas else None
        result = HandResult(
            hand_id=hand_id,
            winner=winner,
            pot=sum(max(d, 0) for d in deltas.values()),
            stack_deltas=deltas,
            board=[_format_card(c) for c in state.board_cards] if state.board_cards else [],
            action_history=action_history,
        )
        self.history.append(result)
        yield Event(type="hand_complete", hand_id=hand_id, payload=result.model_dump())

    def _build_view(
        self,
        hand_id: str,
        ordered: list[str],
        state: State,
        history: list[ActionHistoryEntry],
    ) -> GameView:
        street_idx = max(0, state.street_index or 0)
        street_name = _STREET_NAMES.get(street_idx, "preflop")
        if state.street_index is None:
            street_name = "complete"

        board_cards: list[str] = []
        if state.board_cards:
            for st in state.board_cards:
                board_cards.extend(_format_card(c) for c in st)

        legal: list[LegalAction] = []
        if state.can_fold():
            legal.append(LegalAction(type="fold"))
        if state.can_check_or_call():
            # checking is a call of 0
            call_amt = state.checking_or_calling_amount or 0.0
            if call_amt == 0:
                legal.append(LegalAction(type="check"))
            else:
                legal.append(LegalAction(type="call", amount=float(call_amt)))
        if state.can_complete_bet_or_raise_to():
            min_amt = float(state.min_completion_betting_or_raising_to_amount or 0)
            max_amt = float(state.max_completion_betting_or_raising_to_amount or 0)
            # We expose this as 'raise' in HAB vocabulary (covers bet+raise)
            legal.append(LegalAction(type="raise", amount_min=min_amt, amount_max=max_amt))

        stack_dict = {ordered[i]: float(state.stacks[i]) for i in range(len(ordered))}
        actor = ordered[state.actor_index] if state.actor_index is not None else None
        # current bet to call = call amount facing
        current_bet = float(state.checking_or_calling_amount or 0)
        return GameView(
            hand_id=hand_id,
            table_id="table_1",
            street=street_name,
            board=board_cards,
            pot=float(sum(state.pots) if hasattr(state, "pots") and state.pots else state.total_pot_amount),
            to_act=actor,
            stacks=Stacks(root=stack_dict),
            current_bet=current_bet,
            action_history=history,
            legal_actions=legal,
            big_blind=self.config.big_blind,
            small_blind=self.config.small_blind,
            deadline=datetime.now(timezone.utc).isoformat(),
        )

    def _apply_action(self, state: State, action: Action, legal: list[LegalAction]) -> None:
        if action.action == "fold":
            state.fold()
        elif action.action == "check":
            state.check_or_call()
        elif action.action == "call":
            state.check_or_call()
        elif action.action in ("bet", "raise", "all_in"):
            amt = action.amount or 0
            # pokerkit clamps to legal range
            state.complete_bet_or_raise_to(amt)
```

```python
"""Spawns headless Claude Code subprocesses on demand. Supports a 'mock://' scheme
for tests so we can exercise the orchestrator without burning API credits."""
from __future__ import annotations

import asyncio
import json
import logging
import os
import signal
from pathlib import Path
from typing import Any

from hab.engine.actions import Action

logger = logging.getLogger(__name__)


def _fold(reason: str, hand_id: str | None) -> Action:
    return Action(action="fold", hand_id=hand_id, reason=reason, tool_calls_used=[])


class AgentPool:
    def __init__(
        self,
        shim_url: str,
        max_concurrent: int = 4,
        decision_timeout: float = 300.0,
        claude_binary: str = "claude",
    ):
        self.shim_url = shim_url
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.decision_timeout = decision_timeout
        self.claude_binary = claude_binary
        self.active: dict[str, asyncio.subprocess.Process] = {}

    async def request_action(
        self,
        player_id: str,
        model: str,
        workspace: Path,
        hand_id: str,
    ) -> Action:
        # mock:// short-circuit for tests
        if model.startswith("mock://"):
            return await self._mock_action(model, workspace, hand_id)
        return await self._real_action(player_id, model, workspace, hand_id)

    async def _mock_action(self, model: str, workspace: Path, hand_id: str) -> Action:
        strategy = model[len("mock://"):]
        view_path = workspace / "game_view" / "current_state.json"
        if not view_path.exists():
            return _fold("missing_state", hand_id)
        view = json.loads(view_path.read_text())
        legal = view.get("legal_actions", [])
        if strategy == "always-fold":
            return _fold("mock-strategy", hand_id)
        if strategy == "always-call":
            for la in legal:
                if la["type"] == "check":
                    return Action(action="check", hand_id=hand_id, tool_calls_used=[])
                if la["type"] == "call":
                    return Action(action="call", amount=la["amount"], hand_id=hand_id, tool_calls_used=[])
            return _fold("no-call-available", hand_id)
        if strategy == "min-raise-or-call":
            for la in legal:
                if la["type"] == "raise" and la.get("amount_min"):
                    return Action(action="raise", amount=la["amount_min"], hand_id=hand_id, tool_calls_used=[])
            for la in legal:
                if la["type"] in ("check", "call"):
                    return Action(action=la["type"], amount=la.get("amount"), hand_id=hand_id, tool_calls_used=[])
            return _fold("nothing-legal", hand_id)
        return _fold(f"unknown-mock:{strategy}", hand_id)

    async def _real_action(
        self,
        player_id: str,
        model: str,
        workspace: Path,
        hand_id: str,
    ) -> Action:
        async with self.semaphore:
            (workspace / "actions" / "action.json").unlink(missing_ok=True)
            env = self._build_env(player_id, model)
            prompt = (
                f"You are {player_id}. Read game_view/, decide an action, "
                f"write actions/action.json per CLAUDE.md, then exit. hand_id={hand_id}."
            )
            try:
                proc = await asyncio.create_subprocess_exec(
                    self.claude_binary,
                    "-p", prompt,
                    "--output-format", "stream-json",
                    "--dangerously-skip-permissions",
                    cwd=str(workspace),
                    env=env,
                    stdout=asyncio.subprocess.PIPE,
                    stderr=asyncio.subprocess.PIPE,
                    preexec_fn=os.setsid if os.name != "nt" else None,
                )
            except FileNotFoundError:
                return _fold("claude_binary_missing", hand_id)

            self.active[player_id] = proc
            try:
                _, stderr = await asyncio.wait_for(
                    proc.communicate(), timeout=self.decision_timeout
                )
                if proc.returncode != 0:
                    logger.warning("agent %s exited %s: %s", player_id, proc.returncode, stderr[:500])
            except asyncio.TimeoutError:
                self._kill(proc)
                return _fold("timeout", hand_id)
            except Exception as e:  # noqa: BLE001
                self._kill(proc)
                return _fold(f"error:{e}", hand_id)
            finally:
                self.active.pop(player_id, None)

            af = workspace / "actions" / "action.json"
            if not af.exists():
                return _fold("no_output", hand_id)
            try:
                return Action.model_validate_json(af.read_text())
            except Exception as e:  # noqa: BLE001
                return _fold(f"bad_json:{e}", hand_id)

    def _build_env(self, player_id: str, model: str) -> dict[str, str]:
        return {
            **os.environ,
            "ANTHROPIC_BASE_URL": self.shim_url,
            "ANTHROPIC_AUTH_TOKEN": "hab-internal",
            "ANTHROPIC_MODEL": model,
            "PLAYER_ID": player_id,
        }

    def _kill(self, proc: asyncio.subprocess.Process) -> None:
        try:
            if os.name == "nt":
                proc.kill()
            else:
                os.killpg(os.getpgid(proc.pid), signal.SIGTERM)
        except (ProcessLookupError, PermissionError):
            pass

    async def shutdown(self) -> None:
        for proc in list(self.active.values()):
            self._kill(proc)
        self.active.clear()
```

```python
from __future__ import annotations

import time

from rich.console import Console


class ProgressDisplay:
    def __init__(self, total_hands: int, players: list[str]):
        self.total = total_hands
        self.players = players
        self.console = Console()
        self.start_time = time.time()
        self.hands_done = 0
        self.stacks: dict[str, float] = {}

    def hand_complete(self, hand_id: str, stacks: dict[str, float]):
        self.hands_done += 1
        self.stacks = stacks
        elapsed = time.time() - self.start_time
        eta = elapsed / max(self.hands_done, 1) * max(self.total - self.hands_done, 0)
        line = (
            f"[bold]hand {self.hands_done}/{self.total}[/bold]  "
            + " | ".join(f"{p}: {self.stacks.get(p, 0):.1f}" for p in self.players)
            + f"  | elapsed {elapsed:.0f}s eta {eta:.0f}s"
        )
        self.console.print(line)

    def session_done(self):
        self.console.print(f"[green]session complete:[/green] {self.hands_done} hands "
                           f"in {time.time() - self.start_time:.1f}s")
```

```python
"""HABSession: top-level coordinator.

Starts shim → builds workspaces → drives engine event loop, asking the agent pool
for each action and forwarding it back to the engine."""
from __future__ import annotations

import asyncio
from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path

from hab.engine.game_master import GameMaster, GameMasterConfig
from hab.engine.recorder import HandRecorder
from hab.engine.state import HandResult
from hab.orchestrator.agent_pool import AgentPool
from hab.orchestrator.progress import ProgressDisplay
from hab.orchestrator.workspace_manager import WorkspaceManager
from hab.shim.server import ShimServer


@dataclass
class SessionConfig:
    players: dict[str, str]              # player_id -> model
    hands_target: int = 100
    small_blind: float = 1.0
    big_blind: float = 2.0
    starting_stack: float = 200.0
    output_dir: Path = field(default_factory=lambda: Path.home() / "hab-sessions")
    max_concurrent_agents: int = 4
    decision_timeout_sec: float = 300.0
    seed: int | None = 42
    openrouter_key: str | None = None
    anthropic_key: str | None = None


class HABSession:
    def __init__(self, config: SessionConfig):
        self.config = config
        self.session_id = datetime.now(timezone.utc).strftime("%Y%m%d-%H%M%S")
        self.session_dir = config.output_dir / self.session_id
        self.session_dir.mkdir(parents=True, exist_ok=True)
        self.shim: ShimServer | None = None
        self.pool: AgentPool | None = None
        self.workspaces: WorkspaceManager | None = None
        self.recorder: HandRecorder | None = None
        self.gm: GameMaster | None = None
        self.progress: ProgressDisplay | None = None

    async def run(self) -> dict:
        try:
            await self._startup()
            return await self._main_loop()
        finally:
            await self._cleanup()

    async def _startup(self):
        # Shim only needed when there are non-mock players
        needs_shim = any(not m.startswith("mock://") for m in self.config.players.values())
        if needs_shim:
            self.shim = ShimServer(
                openrouter_key=self.config.openrouter_key,
                anthropic_key=self.config.anthropic_key,
            )
            await self.shim.start()
            shim_url = self.shim.base_url
        else:
            shim_url = "http://localhost:1"  # unused

        self.workspaces = WorkspaceManager(self.session_dir)
        self.recorder = HandRecorder(self.session_dir)
        for pid, model in self.config.players.items():
            self.workspaces.create(pid, model)

        self.pool = AgentPool(
            shim_url=shim_url,
            max_concurrent=self.config.max_concurrent_agents,
            decision_timeout=self.config.decision_timeout_sec,
        )

        gm_cfg = GameMasterConfig(
            players=list(self.config.players.keys()),
            small_blind=self.config.small_blind,
            big_blind=self.config.big_blind,
            starting_stack=self.config.starting_stack,
            hands_target=self.config.hands_target,
            seed=self.config.seed,
            decision_timeout_sec=self.config.decision_timeout_sec,
        )
        self.gm = GameMaster(gm_cfg)
        self.progress = ProgressDisplay(
            total_hands=self.config.hands_target,
            players=list(self.config.players.keys()),
        )

    async def _main_loop(self) -> dict:
        assert self.gm and self.pool and self.workspaces and self.recorder and self.progress
        result_payload: dict = {}
        async for event in self.gm.events():
            if event.type == "action_needed":
                ws = self.workspaces.workspaces_dir / event.player_id
                self.recorder.write_game_view(ws, event.game_view, event.hole_cards)
                self.recorder.reset_action_dir(ws)
                action = await self.pool.request_action(
                    player_id=event.player_id,
                    model=self.config.players[event.player_id],
                    workspace=ws,
                    hand_id=event.hand_id,
                )
                await self.gm.submit_action(event.player_id, action)
            elif event.type == "hand_complete":
                hr = HandResult.model_validate(event.payload)
                self.recorder.write_hand_result(hr)
                stacks = {p: self.gm.stacks[p] for p in self.config.players}
                self.progress.hand_complete(event.hand_id, stacks)
            elif event.type == "session_complete":
                result_payload = event.payload
                self.progress.session_done()
        # Write summary
        summary_path = self.session_dir / "session_summary.json"
        import json
        summary_path.write_text(json.dumps({
            "session_id": self.session_id,
            "players": self.config.players,
            "hands_target": self.config.hands_target,
            **result_payload,
        }, indent=2, default=str))
        return result_payload

    async def _cleanup(self):
        if self.pool:
            await self.pool.shutdown()
        if self.shim:
            await self.shim.stop()
```

```python
"""`hab run <preset>`."""
from __future__ import annotations

import asyncio
import os
from pathlib import Path

import typer
import yaml

from hab.orchestrator.lifecycle import HABSession, SessionConfig


def _load_preset(name: str) -> dict:
    here = Path(__file__).resolve().parent.parent / "presets" / f"{name}.yaml"
    if not here.exists():
        raise typer.BadParameter(f"unknown preset: {name}")
    return yaml.safe_load(here.read_text())


def _load_user_config() -> dict:
    p = Path.home() / ".hab" / "config.yaml"
    if not p.exists():
        return {}
    return yaml.safe_load(p.read_text()) or {}


def run_command(
    preset: str = typer.Argument(...),
    models: str = typer.Option(..., help="comma-separated, e.g. anthropic/claude-opus-4-7,openai/gpt-5"),
    hands: int = typer.Option(None),
    output: Path = typer.Option(None),
    seed: int = typer.Option(42),
):
    preset_data = _load_preset(preset)
    user_cfg = _load_user_config()
    providers = user_cfg.get("providers", {})
    defaults = user_cfg.get("defaults", {})

    or_key = providers.get("openrouter", {}).get("api_key") or os.environ.get("OPENROUTER_API_KEY")
    anth_key = providers.get("anthropic", {}).get("api_key") or os.environ.get("ANTHROPIC_API_KEY")
    out_dir = output or Path(defaults.get("output_dir") or (Path.home() / "hab-sessions")).expanduser()

    model_list = [m.strip() for m in models.split(",") if m.strip()]
    if len(model_list) != 2:
        raise typer.BadParameter("MVP only supports 2 players (heads-up). Pass exactly 2 models.")

    players = {f"player_{chr(ord('a') + i)}": m for i, m in enumerate(model_list)}

    cfg = SessionConfig(
        players=players,
        hands_target=hands or preset_data["session"]["hands_target"],
        small_blind=preset_data["game"]["small_blind"],
        big_blind=preset_data["game"]["big_blind"],
        starting_stack=preset_data["game"]["starting_stack"],
        output_dir=out_dir,
        max_concurrent_agents=defaults.get("max_concurrent_agents", 4),
        decision_timeout_sec=defaults.get("decision_timeout_sec", 300),
        seed=seed,
        openrouter_key=or_key,
        anthropic_key=anth_key or None,
    )

    needs_real = any(not m.startswith("mock://") for m in model_list)
    if needs_real and not or_key:
        typer.echo("❌ OPENROUTER_API_KEY not set. Run `hab init` or set the env var.")
        raise typer.Exit(2)

    session = HABSession(cfg)
    typer.echo(f"📂 Session dir: {session.session_dir}")
    result = asyncio.run(session.run())
    typer.echo(f"\n✅ Done. Hands played: {result.get('hands_played')}")
    typer.echo(f"   Final stacks: {result.get('final_stacks')}")
```

```python
from __future__ import annotations

import asyncio

from pathlib import Path

import pytest

from hab.orchestrator.lifecycle import HABSession, SessionConfig


@pytest.mark.asyncio
async def test_e2e_mock_vs_mock_100_hands(tmp_path: Path):
    cfg = SessionConfig(
        players={
            "player_a": "mock://always-fold",
            "player_b": "mock://always-call",
        },
        hands_target=100,
        small_blind=1.0,
        big_blind=2.0,
        starting_stack=200.0,
        output_dir=tmp_path,
        max_concurrent_agents=2,
        decision_timeout_sec=10.0,
        seed=42,
    )
    session = HABSession(cfg)
    result = await asyncio.wait_for(session.run(), timeout=120)

    # Mock A always folds preflop, so B should win on most hands until A busts.
    assert result["hands_played"] >= 1
    final = result["final_stacks"]
    assert sum(final.values()) == pytest.approx(400.0, rel=1e-3)

    # Hand history JSONs written
    hand_files = list((session.session_dir / "hands").glob("*.json"))
    assert len(hand_files) == result["hands_played"]
    sample = json.loads(hand_files[0].read_text())
    assert "stack_deltas" in sample
```

## Architecture (one-liner)

`hab` launches a single in-process orchestrator that runs a FastAPI shim
(translating Anthropic ↔ OpenAI for OpenRouter), drives a pokerkit engine,
and forks a `claude -p` subprocess per decision. Memory is on disk; agents
are stateless processes.

## What works today

- `hab run quickstart` heads-up 100 hands
- `mock://always-fold` / `mock://always-call` for free local testing
- pokerkit-backed NLHE engine with action validation
- Subprocess agent pool with process-group isolation and timeout-fold

## What's coming

- MCP toolkit (equity / GTO / opponent DB / notes)
- 6-max + duplicate poker
- Three-layer scoring (Raw BB/100, Skill BB/100, Elo)
- GitHub Pages leaderboard

See `docs/design/development-plan-v2.2.md` for the full design.
